系统管理

MAG服务的启动、状态监控和关闭管理

mag.start()

启动MAG服务器和MCP客户端。这个函数会启动必要的后台服务，使SDK可以正常工作。每个Python进程只需要调用一次。

函数签名

def start() -> bool:
    """
    启动MAG服务器和MCP客户端
    
    返回:
        bool: 是否成功启动服务
    """

返回值

返回值说明

bool - 返回 True 表示服务成功启动，False 表示启动失败

使用示例

import mag

# 启动MAG服务
success = mag.start()

if success:
    print("MAG服务启动成功！")
    # 现在可以使用其他MAG功能
else:
    print("MAG服务启动失败！")

注意事项

• 服务启动可能需要几秒钟时间
• 如果服务已经运行，重复调用start()会直接返回True
• 启动失败通常是因为端口被占用或权限不足

mag.is_running()

检查MAG服务是否正在运行。用于确认服务状态或在调用其他功能前进行状态检查。

函数签名

def is_running() -> bool:
    """
    检查MAG服务是否正在运行
    
    返回:
        bool: 服务是否正在运行
    """

返回值

返回值说明

bool - 返回 True 表示服务正在运行，False 表示服务未运行

使用示例

import mag

# 检查服务状态
if mag.is_running():
    print("MAG服务正在运行")
    
    # 可以安全地调用其他MAG功能
    graphs = mag.list_graphs()
    print(f"可用图列表: {graphs}")
else:
    print("MAG服务未运行，正在启动...")
    if mag.start():
        print("服务启动成功！")
    else:
        print("服务启动失败！")

实际应用示例

import mag
import time

def ensure_service_running():
    """确保MAG服务正在运行"""
    if not mag.is_running():
        print("正在启动MAG服务...")
        if mag.start():
            print("服务启动成功！")
            return True
        else:
            print("服务启动失败！")
            return False
    else:
        print("MAG服务已经在运行")
        return True

# 在使用MAG功能前先确保服务运行
if ensure_service_running():
    # 执行你的MAG相关操作
    models = mag.list_models()
    print(f"可用模型: {len(models)} 个")
mag.shutdown()

关闭MAG服务器和MCP客户端。这个函数会安全地终止所有后台进程，清理资源。

函数签名

def shutdown() -> None:
    """
    关闭MAG服务器和MCP客户端
    
    这个函数会安全地终止所有后台进程。
    """

返回值

返回值说明

None - 无返回值，函数执行完成表示关闭操作已完成

使用示例

import mag

# 启动服务并执行一些操作
mag.start()

try:
    # 执行一些MAG操作
    graphs = mag.list_graphs()
    print(f"处理了 {len(graphs)} 个图")
    
    # ... 其他操作 ...
    
finally:
    # 确保在程序结束前关闭服务
    print("正在关闭MAG服务...")
    mag.shutdown()
    print("服务已关闭")

优雅关闭示例

import mag
import atexit
import signal
import sys

def cleanup_on_exit():
    """程序退出时的清理函数"""
    if mag.is_running():
        print("检测到程序退出，正在关闭MAG服务...")
        mag.shutdown()
        print("MAG服务已安全关闭")

def signal_handler(signum, frame):
    """处理中断信号"""
    print(f"收到信号 {signum}，正在优雅退出...")
    cleanup_on_exit()
    sys.exit(0)

# 注册退出清理函数
atexit.register(cleanup_on_exit)

# 注册信号处理器
signal.signal(signal.SIGINT, signal_handler)   # Ctrl+C
signal.signal(signal.SIGTERM, signal_handler)  # 终止信号

# 启动MAG服务
if mag.start():
    print("MAG服务已启动，可以安全执行操作...")
    
    # 你的主要逻辑
    try:
        while True:
            # 执行你的长期运行任务
            time.sleep(1)
    except KeyboardInterrupt:
        print("用户中断程序")

重要提示

• 建议在程序结束前总是调用shutdown()以确保资源被正确清理
• 服务关闭后，需要重新调用start()才能使用MAG功能
• 可以使用atexit模块确保程序异常退出时也能正确关闭服务

完整工作流示例

下面是一个完整的系统管理工作流示例，展示了如何正确地启动、使用和关闭MAG服务：

import mag
import time

class MAGManager:
    """MAG服务管理器"""
    
    def __init__(self):
        self.service_running = False
    
    def start_service(self):
        """启动MAG服务"""
        print("正在启动MAG服务...")
        
        if mag.is_running():
            print("MAG服务已经在运行")
            self.service_running = True
            return True
        
        success = mag.start()
        if success:
            self.service_running = True
            print("✅ MAG服务启动成功！")
            
            # 验证服务状态
            time.sleep(2)  # 等待服务完全启动
            if mag.is_running():
                print("✅ 服务状态验证通过")
                return True
            else:
                print("❌ 服务状态验证失败")
                return False
        else:
            print("❌ MAG服务启动失败！")
            return False
    
    def check_service_health(self):
        """检查服务健康状态"""
        if not self.service_running:
            return False
            
        try:
            # 尝试调用一个简单的API来测试服务
            mag.list_graphs()  # 这是一个轻量级的检查
            return True
        except Exception as e:
            print(f"服务健康检查失败: {e}")
            return False
    
    def stop_service(self):
        """停止MAG服务"""
        if self.service_running:
            print("正在关闭MAG服务...")
            mag.shutdown()
            self.service_running = False
            print("✅ MAG服务已关闭")
    
    def __enter__(self):
        """上下文管理器入口"""
        if self.start_service():
            return self
        else:
            raise RuntimeError("无法启动MAG服务")
    
    def __exit__(self, exc_type, exc_val, exc_tb):
        """上下文管理器出口"""
        self.stop_service()

# 使用示例
def main():
    # 方式1: 使用上下文管理器（推荐）
    try:
        with MAGManager() as mag_manager:
            print("开始执行MAG相关任务...")
            
            # 检查服务健康状态
            if mag_manager.check_service_health():
                print("服务运行正常，开始执行任务")
                
                # 执行你的MAG操作
                graphs = mag.list_graphs()
                models = mag.list_models()
                
                print(f"发现 {len(graphs)} 个图和 {len(models)} 个模型")
                
                # 更多操作...
                
            else:
                print("服务健康检查失败")
                
    except Exception as e:
        print(f"执行过程中出现错误: {e}")
    
    print("程序执行完成")

if __name__ == "__main__":
    main()
